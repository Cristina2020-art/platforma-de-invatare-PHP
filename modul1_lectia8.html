Lucrul cu baze de date MySQL în PHP

Acasă | Cursuri | Modul 1

Conectarea la baza de date MySQL

<?php
$host = "localhost";
$user = "utilizator";
$parola = "parola";
$baza_date = "numele_bazei";

$conn = new mysqli($host, $user, $parola, $baza_date);

if ($conn->connect_error) {
  die("Conexiune eșuată: " . $conn->connect_error);
}
echo "Conexiune realizată cu succes!";
?>

Executarea unei interogări SELECT

<?php
$sql = "SELECT id, nume FROM utilizatori";
$result = $conn->query($sql);

if ($result->num_rows > 0) {
  while($row = $result->fetch_assoc()) {
    echo "ID: " . $row["id"]. " - Nume: " . $row["nume"]. "<br>";
  }
} else {
  echo "0 rezultate";
}
$conn->close();
?>

Inserarea datelor în tabel

<?php
$sql = "INSERT INTO utilizatori (nume, email) VALUES ('Ion', 'ion@example.com')";

if ($conn->query($sql) === TRUE) {
  echo "Înregistrare adăugată cu succes.";
} else {
  echo "Eroare: " . $conn->error;
}
?>
