Clase abstracte și interfețe

<?php
abstract class Animal {
  abstract public function sunet();
}

class Caine extends Animal {
  public function sunet() {
    return "Ham Ham";
  }
}

interface Miscare {
  public function merge();
}

class Om implements Miscare {
  public function merge() {
    return "Merg pe jos";
  }
}
?>

Trait-uri

<?php
trait Mesaj {
  public function salut() {
    return "Salut!";
  }
}

class Utilizator {
  use Mesaj;
}
?>
