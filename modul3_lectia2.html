Singleton

<?php
class Singleton {
  private static $instanta = null;

  private function __construct() {}

  public static function getInstanta() {
    if (self::$instanta === null) {
      self::$instanta = new Singleton();
    }
    return self::$instanta;
  }
}
?>

Factory

<?php
interface Animal {
  public function sunet();
}

class Caine implements Animal {
  public function sunet() {
    return "Ham Ham";
  }
}

class Pisica implements Animal {
  public function sunet() {
    return "Miau";
  }
}

class FactoryAnimal {
  public static function create($tip) {
    if ($tip === "caine") {
      return new Caine();
    } elseif ($tip === "pisica") {
      return new Pisica();
    }
    throw new Exception("Tip necunoscut");
  }
}
?>

Observer

<?php
interface Observer {
  public function update($mesaj);
}

class Subscriber implements Observer {
  public function update($mesaj) {
    echo "Mesaj primit: $mesaj";
  }
}

class Publisher {
  private $observers = [];

  public function attach(Observer $obs) {
    $this->observers[] = $obs;
  }

  public function notify($mesaj) {
    foreach ($this->observers as $obs) {
      $obs->update($mesaj);
    }
  }
}
?>
