Structuri de control în PHP

Acasă | Cursuri | Modul 1

Instrucțiuni condiționale (if - else)

<?php
$nota = 7;

if ($nota >= 5) {
  echo "Ai trecut.";
} else {
  echo "Nu ai trecut.";
}
?>

Instrucțiunea elseif

<?php
$nota = 7;

if ($nota >= 9) {
  echo "Excelent!";
} elseif ($nota >= 5) {
  echo "Bine!";
} else {
  echo "Încearcă din nou.";
}
?>

Switch

<?php
$zi = "luni";

switch ($zi) {
  case "luni":
    echo "Începe săptămâna.";
    break;
  case "vineri":
    echo "Vine weekendul!";
    break;
  default:
    echo "Altă zi.";
}
?>

Bucla for

<?php
for ($i = 1; $i <= 5; $i++) {
  echo "Număr: $i <br>";
}
?>

Bucla while

<?php
$i = 1;
while ($i <= 5) {
  echo "Număr: $i <br>";
  $i++;
}
?>
